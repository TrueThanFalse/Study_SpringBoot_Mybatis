// board Table 생성
CREATE TABLE board(
bno BIGINT AUTO_INCREMENT,
title VARCHAR(200) NOT NULL,
writer VARCHAR(200) NOT NULL,
content TEXT NOT NULL,
reg_at DATETIME DEFAULT NOW(),
mod_at DATETIME DEFAULT NOW(),
PRIMARY KEY(bno)
);

CREATE TABLE comment(
cno BIGINT AUTO_INCREMENT,
bno BIGINT NOT NULL,
writer VARCHAR(200) NOT NULL,
content TEXT NOT NULL,
reg_at DATETIME DEFAULT NOW(),
mod_at DATETIME DEFAULT NOW(),
PRIMARY KEY(cno)
);
